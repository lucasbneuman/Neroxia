# Comandos para ejecutar tests de API en PowerShell

# Ya estás en: C:\Users\avali\Desktop\Proyectos\whatsapp_sales_bot\apps\api
# NO necesitas hacer cd de nuevo

# 1. Instalar pytest-cov (necesario para las opciones de coverage en pytest.ini)
pip install pytest-cov

# 2. Ejecutar tests unitarios
pytest tests/unit/ -v

# 3. Ejecutar tests de integración
pytest tests/integration/ -v

# 4. Ejecutar todos los tests
pytest tests/ -v

# 5. Ejecutar con coverage (después de instalar pytest-cov)
pytest tests/ -v --cov=src --cov-report=html

# 6. Ver reporte de coverage
start htmlcov/index.html

# ALTERNATIVA: Si no quieres instalar pytest-cov, ejecuta sin coverage:
pytest tests/unit/ -v --no-cov
pytest tests/integration/ -v --no-cov
